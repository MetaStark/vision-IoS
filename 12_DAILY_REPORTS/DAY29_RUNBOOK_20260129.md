WITH ranked AS (
    SELECT
        RANK() OVER (ORDER BY pre_tier_score_at_birth) as score_rank,

    AND status = 'FALSIFIED'

)
SELECT ROUND(CORR(score_rank, ttf_rank)::numeric, 4) as spearman_rho FROM ranked;
